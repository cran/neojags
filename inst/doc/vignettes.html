Parameter Estimation

Create model for JAGS

model_string <- "
model {
  # Likelihood
  for (i in 1:100) {
    x[i] ~ djskew.ep(mu, tau,nu1, nu2)
  }
  
  # Prior distributions
  mu ~ dnorm(2,10000)
  tau ~ dgamma(10,10)
  nu1 ~ dgamma(10,10)
  nu2 ~ dgamma(10,10)
}
"

Compile the model

model <- jags.model(textConnection(model_string), data = list(x=c(x)),n.chains=2)
#> Compiling model graph
#>    Resolving undeclared variables
